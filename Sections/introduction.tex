\begin{frame}{Because POs Are Versatile}
    \begin{minipage}{0.45\textwidth}
        \textbf{PP (Polypropylene)}
        \begin{itemize}
            \item $M_w$
            \item Tacticity
            
        \end{itemize}
    \end{minipage}
    \hfill
    \begin{minipage}{0.45\textwidth}
        \textbf{PE (Polyethylene)}
        \begin{itemize}
            \item $M_w$
            \item PDI
            \item Branching 
        \end{itemize}
    \end{minipage}
    \begin{block}{for thermoplastics}
        $M_w + \text{ PDI } + \text{ branching } + \text{ Tacticity } = \chi_c$ \& properties ($E$,$\rho$, $\sigma$, $\mu$  and $T_g$) 
    \end{block}
    \begin{alertblock}{trade off}
    PE density $\rightarrow$ $\chi_c$ $\rightarrow$ HDPE pipes vs LDPE films \\
    PP tacticity $\rightarrow$ $\chi_c$ $\rightarrow$ PP $\rho$ 
    \end{alertblock}
    % high M_w lower the kinetic of formation of lamelae and the cristallinity rate but higher the sterngth due to entaglement 
    % high PDI lower the cristallinity and can act as local defect, lead to complex viscosity
    % high branching lower the cristallinity rate 
    % High tacticity propote good cristallinity rate.
\end{frame}

\begin{frame}{Summary of PP/PE properties}
    \begin{minipage}{0.48\textwidth}
        \textbf{PE (Polyethylene) \cite{hughes_polyethylene_2023, specialchem_tg_plastics, wiki_glass_transition}}
        \vspace{0.2cm}
        \begin{block}{Properties}
            \begin{itemize}
                \item Low density: 0.91--0.97 g/cm$^3$
                \item Chemical resistance: Excellent 
                \item $T_g$ = -120°C to -110°C
                \item $T_m$ = 105 to 136°C 
                \item $\chi_c$ = 0.4-0.9
                \item Temp range: -100°C to +130°C (HDPE), +80°C continuous
                \item Cost: \$1,200--1,800/tonne (2026 avg)
            \end{itemize}
        \end{block}
    \end{minipage}%
    \hfill
    \begin{minipage}{0.48\textwidth}
        \textbf{PP (Polypropylene) \cite{hughes_polyethylene_2023, specialchem_tg_plastics, wiki_glass_transition}}

        \vspace{0.2cm}
        \begin{block}{Key Advantages}
            \begin{itemize}
                \item Low density: 0.90--0.91 g/cm$^3$ 
                \item Chemical resistance: Excellent (acids, bases, oils)
                \item $T_g$ = -120°C to -110°C
                \item $T_m$ = 150 to 170°C 
                \item $\chi_c$ = 0.3-0.6
                \item Temp range: 0°C to +140°C
                \item Cost: \$1,100--1,600/tonne (2026 avg)
            \end{itemize}
        \end{block}
    \end{minipage}
\end{frame}
